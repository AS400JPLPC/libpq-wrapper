200  template <typename Head, typename... Args>
201  std::string prepare(std::string const& format, Head&& head, Args&&... args)
202  {
203 
204  if (count_format_specifiers(format) > 0 && first == true)
205  {
206  fin_stage = format ; first = false ; sqlstream.str("");
207  auto last_format_pos = format.find_last_of(FORMAT_SPECIFIER);
208  fin_stage =fin_stage.erase(0, last_format_pos +1);
209  }
210 
211  if (count_format_specifiers(format) != sizeof...(Args) + 1)
212  throw std::invalid_argument("number of arguments does not match the format string");
213 
214  // TODO: take care of escaped format specifiers
215  sqlstream <<"";
216  auto first_format_pos = format.find_first_of(FORMAT_SPECIFIER);
217 
218  sqlstream << format.substr(0, first_format_pos);
219  sqlstream << head;
220 
221  if ( count_format_specifiers(format) == 1 )
222  {
223  sqlstream<<fin_stage; first =true ;
224  }
225 
226  prepare(format.substr(first_format_pos+1), std::forward<Args>(args)...);
227  return sqlstream.str();
228 
229  }
