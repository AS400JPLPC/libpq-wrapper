224  {

248 std::string libPQwrp::prepare(std::string const& format, Head&& head, Args&&... args)
249 {

257

259 
260  if (count_format_specifiers(format) != sizeof...(Args) + 1)
261  throw std::invalid_argument("number of arguments does not match the format string");
262 
263  // TODO: take care of escaped format specifiers
264  sqlstream <<"";
265  auto first_format_pos = format.find_first_of(FORMAT_SPECIFIER);
266 
267  sqlstream << format.substr(0, first_format_pos);
268  sqlstream << head;
269 
270  if ( count_format_specifiers(format) == 1 ) { sqlstream<<fin_stage; first =true ;}
271 
272  prepare(format.substr(first_format_pos+1), std::forward<Args>(args)...);
